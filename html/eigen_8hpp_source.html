19  // Determine coefficients of characteristic poynomial. We write
20  // | a d f |
21  // A = | d* b e |
22  // | f* e* c |
23  float de = get<0, 1>(aTensor) * get<1, 2>(aTensor); // d * e
24  float dd = sqr(get<0,1>(aTensor)); // d^2
25  float ee = sqr(get<1,2>(aTensor)); // e^2
26  float ff = sqr(get<0,2>(aTensor)); // f^2
27  float m = matrix_trace(aTensor);
28  float c1 =
29  get<0,0>(aTensor) * get<1,1>(aTensor) +
30  get<0,0>(aTensor) * get<2,2>(aTensor) +
31  get<1,1>(aTensor) * get<2,2>(aTensor) -
32  (dd + ee + ff); // a*b + a*c + b*c - d^2 - e^2 - f^2
33  float c0 =
34  get<2,2>(aTensor) * dd +
35  get<0,0>(aTensor) * ee +
36  get<1,1>(aTensor) * ff -
37  get<0,0>(aTensor) * get<1,1>(aTensor)*get<2,2>(aTensor) -
38  2.0 * get<0,2>(aTensor) * de; // c*d^2 + a*e^2 + b*f^2 - a*b*c - 2*f*d*e)
39 
40  float p = sqr(m) - 3.0 * c1;
41  float q = m * (p - (3.0/2.0) * c1) - (27.0/2.0) * c0;
42  float sqrt_p = sqrt(fabs(p));
43 
44  float phi = 27.0 * (0.25 * sqr(c1) * (p - c1) + c0*(q + 27.0/4.0 * c0));
45  phi = (1.0 / 3.0) * atan2(sqrt(fabs(phi)), q);
46 
47  float c = sqrt_p * cos(phi);
48  float s = (1.0/M_SQRT3) * sqrt_p * sin(phi);
49 
50  simple_vector<float, 3> vals;
51  vals[1] = (1.0/3.0) * (m - c);
52  vals[2] = vals[1] + s;
53  vals[0] = vals[1] + c;
54  vals[1] -= s;
55  return vals;
56 }

64  auto v2 = aMatrix.column(tCol2);
65  v2[tCol2] -= aEigenValue;

67  auto n2 = squared_magnitude(v2);
68  if (n1 <= aThreshold) {
69  return simple_vector<float, 3>(1.0f, 0.0f, 0.0f);
70  } else if (n2 <= aThreshold) {
71  return simple_vector<float, 3>(0.0f, 1.0f, 0.0f);
72  }
73  auto result = cross(v1, v2);
74  auto norm = squared_magnitude(result);
75  auto threshold2 = sqr(64.0f * EPSILON) * n1 * n2;
76  if (norm < threshold2) {
77  // If angle between A[0] and A[1] is too small, don't use
78  // cross product, but calculate v ~ (1, -A0/A1, 0)
79  auto t = sqr(get<0, 1>(aMatrix));
80  auto f = -get<0, 0>(aMatrix) / get<0, 1>(aMatrix);
81  if (sqr(get<1, 1>(aMatrix) > t)) {
82  t = sqr(get<1, 1>(aMatrix));
83  f = -get<0, 1>(aMatrix) / get<1, 1>(aMatrix);
84  }
85  if (sqr(get<1, 2>(aMatrix) > t)) {
86  f = -get<0, 2>(aMatrix) / get<1, 2>(aMatrix);
87  }
88  norm = 1.0f / sqrt(1.0f + sqr(f));
89  return simple_vector<float, 3>(norm, f * norm, 0.0f);
90  }
