762 find_orthonormal(const TVector &aVector)
763 {
764  simple_vector<typename static_vector_traits<TVector>::element_type, 3> result;
765  for (int j = 0; j < 3; ++j) {
766  // find some orthogonal vector to the first eigen vector
767  if (aVector[j] != 0.0f) {
768  // swap non-zero coordinate with following one and clear the third -> perpendicular vector
769  auto norm = 1.0f / sqrt(sqr(aVector[j]) + sqr(aVector[(j + 1) % 3]));
770  result[j] = aVector[(j + 1) % 3] * norm;
771  result[(j + 1) % 3] = -aVector[j] * norm;
772  result[(j + 2) % 3] = 0.0f;
773  return result;
774  }
775  }
776  return result;
